// ============================================================ // Supabase Edge Function: trial-reminder // Envía un email a los usuarios cuya prueba de 7 días termina // pronto (día 5 y último día), para recuperar abandonos. // ------------------------------------------------------------ // Cómo funciona: // - Se ejecuta sola cada día (pg_cron, ver trial_reminder_cron.sql). // - Busca en user_state las pruebas que vencen en ~2 días y mañana. // - Envía el correo con Resend (https://resend.com — 3.000 emails/mes gratis). // // Desplegar: supabase functions deploy trial-reminder --no-verify-jwt // Secrets: // RESEND_API_KEY = re_... (tu clave de Resend) // RESEND_FROM = CourtLab (remitente verificado en Resend) // CRON_SECRET = una-cadena-larga-secreta (para que solo el cron la llame) // // Requiere en user_state una columna 'trial_ends_at timestamptz' // (la rellena el webhook de Stripe al iniciar la prueba). Ver // trial_reminder_cron.sql para crearla y programar el cron. // ============================================================ import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'; const RESEND_API_KEY = Deno.env.get('RESEND_API_KEY')!; const RESEND_FROM = Deno.env.get('RESEND_FROM') ?? 'CourtLab '; const CRON_SECRET = Deno.env.get('CRON_SECRET') ?? ''; const APP_URL = Deno.env.get('APP_URL') ?? 'https://tenislab.github.io/TennisLab'; async function sendEmail(to: string, subject: string, html: string) { const r = await fetch('https://api.resend.com/emails', { method: 'POST', headers: { 'Authorization': `Bearer ${RESEND_API_KEY}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ from: RESEND_FROM, to, subject, html }), }); return r.ok; } function body(daysLeft: number) { const last = daysLeft <= 1; return `

CourtLab

${last ? 'Tu prueba Premium termina hoy. Consérvala para seguir con tácticas ilimitadas, vídeo/GIF, informes PDF y Modo Club.' : 'Te quedan 2 días de prueba Premium. ¿Sigues con todo desbloqueado?'}

Abrir CourtLab

Si no quieres continuar, no tienes que hacer nada: la prueba se cancela sola.

`; } Deno.serve(async (req) => { // protección: solo el cron (con el secreto) puede dispararla const auth = req.headers.get('x-cron-secret') || new URL(req.url).searchParams.get('secret'); if (CRON_SECRET && auth !== CRON_SECRET) { return new Response(JSON.stringify({ error: 'forbidden' }), { status: 403 }); } try { const admin = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!); const now = Date.now(); const d2 = new Date(now + 2 * 86400000), d2b = new Date(now + 3 * 86400000); // ventana día 5 (~2 días para vencer) const d1 = new Date(now), d1b = new Date(now + 1 * 86400000); // último día // candidatos: prueba que vence dentro de la ventana y aún no avisados de esa fase const { data: rows } = await admin.from('user_state') .select('user_id, trial_ends_at, trial_reminded') .not('trial_ends_at', 'is', null) .eq('is_premium', false); let sent = 0; for (const row of rows || []) { const ends = new Date(row.trial_ends_at).getTime(); let phase = 0; if (ends >= d2.getTime() && ends < d2b.getTime()) phase = 5; else if (ends >= d1.getTime() && ends < d1b.getTime()) phase = 7; if (!phase) continue; if ((row.trial_reminded || 0) >= phase) continue; // ya avisado de esta fase // email del usuario (desde auth) const { data: u } = await admin.auth.admin.getUserById(row.user_id); const email = u?.user?.email; if (!email) continue; const daysLeft = phase === 7 ? 1 : 2; const ok = await sendEmail(email, daysLeft <= 1 ? 'Tu prueba Premium termina hoy' : 'Te quedan 2 días de prueba Premium', body(daysLeft)); if (ok) { await admin.from('user_state').update({ trial_reminded: phase }).eq('user_id', row.user_id); sent++; } } return new Response(JSON.stringify({ ok: true, sent }), { headers: { 'Content-Type': 'application/json' } }); } catch (e) { return new Response(JSON.stringify({ error: String(e) }), { status: 500 }); } });
