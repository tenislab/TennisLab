// ============================================================ // Supabase Edge Function: stripe-webhook // Stripe avisa aquí cuando alguien paga / cancela. Marcamos // is_premium en la tabla user_state del usuario correspondiente. // ------------------------------------------------------------ // Desplegar (sin verificación de JWT, la hace la firma de Stripe): // supabase functions deploy stripe-webhook --no-verify-jwt // Secrets: // STRIPE_SECRET_KEY = sk_live_... / sk_test_... // STRIPE_WEBHOOK_SECRET = whsec_... (lo da Stripe al crear el webhook) // SUPABASE_URL = (automático) // SUPABASE_SERVICE_ROLE_KEY= service_role key (Settings → API) ← NUNCA en el cliente // En Stripe → Developers → Webhooks → Add endpoint: // URL: https://.supabase.co/functions/v1/stripe-webhook // Eventos: checkout.session.completed, customer.subscription.deleted, // customer.subscription.updated // ============================================================ import Stripe from 'https://esm.sh/stripe@14?target=deno'; import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'; const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY')!, { apiVersion: '2023-10-16' }); const WH_SECRET = Deno.env.get('STRIPE_WEBHOOK_SECRET')!; const admin = createClient( Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!, // service role: salta RLS (solo en servidor) ); async function setPremium(userId: string, on: boolean, extra: Record = {}) { if (!userId) return; await admin.from('user_state') .upsert({ user_id: userId, is_premium: on, updated_at: new Date().toISOString(), ...extra }); } Deno.serve(async (req) => { const sig = req.headers.get('stripe-signature')!; const body = await req.text(); let event: Stripe.Event; try { event = await stripe.webhooks.constructEventAsync(body, sig, WH_SECRET); } catch (e) { return new Response(`bad signature: ${e}`, { status: 400 }); } switch (event.type) { case 'checkout.session.completed': { const s = event.data.object as Stripe.Checkout.Session; const userId = (s.client_reference_id || s.metadata?.user_id) as string; await setPremium(userId, true, { stripe_customer_id: s.customer as string, stripe_subscription_id: s.subscription as string, }); break; } case 'customer.subscription.updated': { const sub = event.data.object as Stripe.Subscription; const active = sub.status === 'active' || sub.status === 'trialing'; await setPremium(sub.metadata?.user_id as string, active); break; } case 'customer.subscription.deleted': { const sub = event.data.object as Stripe.Subscription; await setPremium(sub.metadata?.user_id as string, false); break; } } return new Response('ok', { status: 200 }); });
